// Kernel ATR calculation using kernel-smoothed high/low/close
kernel_atr(int length, float _high, float _low, float _close) =>
    float trueRange = na(_high[1]) ? _high - _low : math.max(math.max(_high - _low, math.abs(_high - _close[1])), math.abs(_low - _close[1]))
    ta.rma(trueRange, length)

// Calculate NWE envelope bounds
// Returns: [yhat, upper_near, upper_far, lower_near, lower_far]
calcNWE(float _close, float _high, float _low, int h, float alpha, int x0, int atrLen, float nearF, float farF) =>
    // Kernel regression on price
    float yhat_close = kernels.rationalQuadratic(_close, h, alpha, x0)
    float yhat_high = kernels.rationalQuadratic(_high, h, alpha, x0)
    float yhat_low = kernels.rationalQuadratic(_low, h, alpha, x0)

    // Kernel ATR
    float ktr = kernel_atr(atrLen, yhat_high, yhat_low, yhat_close)

    // Envelope bounds
    float upper_far = yhat_close + farF * ktr
    float upper_near = yhat_close + nearF * ktr
    float lower_near = yhat_close - nearF * ktr
    float lower_far = yhat_close - farF * ktr

    [yhat_close, upper_near, upper_far, lower_near, lower_far]

// Scan 100 bars for Order Blocks in 4 categories:
// 1. Unmitigated Bullish OB: formed with gap+sweep, not broken since (valid demand zone)
// 2. Unmitigated Bearish OB: formed with gap+sweep, not broken since (valid supply zone)
// 3. Breaker Resistance: Bullish OB where close broke below OB's low (broken demand → resistance)
// 4. Breaker Support: Bearish OB where close broke above OB's high (broken supply → support)
//
// Returns: [unmitBullBar, unmitBearBar, brkResBar, brkSupBar]
// Each value = bars ago of nearest OB in that category (0 = none found)
//
// OB FORMATION (from OB & FVG.txt):
// An OB forms when TWO conditions are met at the same time:
// 1. SWEEP: The OB candle swept liquidity (lows for bullish, highs for bearish)
// 2. GAP: There's a gap between OB candle and validation candle (2 bars later)
//
// Bar structure at formation:
//   bar[i]   = OB candle (where sweep happened)
//   bar[i-1] = intermediate candle
//   bar[i-2] = validation candle (creates the gap)
//
// OB VALIDITY:
// - An OB stays valid until price CLOSES through its level
// - Bullish OB invalid when: any close < OB's low (becomes breaker resistance)
// - Bearish OB invalid when: any close > OB's high (becomes breaker support)
// - FVG fill is SEPARATE from OB validity (gap can close but OB still valid)

// Main scan function (non-repainting, uses confirmed bars)
// Returns: [unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar]
// - unmitBullBar: Unmitigated bullish OB (pristine, never touched)
// - unmitBearBar: Unmitigated bearish OB (pristine, never touched)
// - mitBullBar: Mitigated bullish OB (tested but not broken)
// - mitBearBar: Mitigated bearish OB (tested but not broken)
// - brkResBar: Breaker resistance (bullish OB broken down)
// - brkSupBar: Breaker support (bearish OB broken up)
scanOBRange() =>
    int unmitBullBar = 0
    int unmitBearBar = 0
    int mitBullBar = 0
    int mitBearBar = 0
    int brkResBar = 0
    int brkSupBar = 0

    // Start at 4 to have room for:
    // - bar[i+1] for sweep check (older adjacent bar)
    // - bar[i-3] for bar[3] gap pattern validation
    // Scan up to 100 bars back
    for i = 4 to 100
        //////////////////////////////////////////
        // BULLISH OB DETECTION (Demand Zone)
        //////////////////////////////////////////

        // Step 1: Check SWEEP - OB candle swept lows of adjacent bars
        bool bullSweep = low[i] < low[i-1] and low[i] < low[i+1]

        // Step 2: Check GAP AT FORMATION
        // Original indicator checks both bar[2] and bar[3] patterns:
        // - bar[2] pattern: validation bar is 2 bars after OB → high[i] < low[i-2]
        // - bar[3] pattern: validation bar is 3 bars after OB → high[i] < low[i-3]
        bool bullGap2 = high[i] < low[i-2]  // bar[2] formation pattern
        bool bullGap3 = high[i] < low[i-3]  // bar[3] formation pattern
        bool bullGapAtFormation = bullGap2 or bullGap3

        // OB formed if BOTH sweep AND gap conditions were true
        bool bullOBFormed = bullSweep and bullGapAtFormation

        if bullOBFormed
            float obHigh = high[i]
            float obLow = low[i]

            // Calculate test price (30% into OB from high)
            float testPrice = obHigh - ((obHigh - obLow) * ob_testPercent / 100)

            // Step 3: Check OB lifecycle - Active → Mitigated → Breaker → Reversed
            // Loop from formation to current bar, tracking state changes
            bool isBroken = false
            bool isReversed = false
            bool hasReachedTestLevel = false
            bool isTested = false
            int startBar = bullGap3 ? i - 4 : i - 3  // Start after validation bar
            int j = startBar
            while j >= 1
                if not isBroken
                    // Check if became BREAKER (close below OB's low)
                    if close[j] < obLow
                        isBroken := true
                    else
                        // Check mitigation (two-step process from original indicator)
                        // Step 1: Price reaches test level (dipped into OB zone)
                        if not hasReachedTestLevel and low[j] <= testPrice
                            hasReachedTestLevel := true
                        // Step 2: After reaching test level, check if price returns/bounces
                        if hasReachedTestLevel and not isTested
                            if low[j] >= obLow or (low[j] < obLow and high[j] >= obLow)
                                isTested := true
                else if not isReversed
                    // Check if became REVERSED (close back above OB's high)
                    if close[j] > obHigh
                        isReversed := true
                j := j - 1

            // Categorize based on final state (priority: reversed > broken > mitigated > unmitigated)
            if isReversed
                // Skip - reversed OBs are no longer valid
                na(close)  // no-op: skip reversed OBs
            else if isBroken
                // It's a BREAKER RESISTANCE (demand zone that failed)
                if brkResBar == 0
                    brkResBar := i
            else if isTested
                // It's MITIGATED (tested but not broken)
                if mitBullBar == 0
                    mitBullBar := i
            else
                // It's still UNMITIGATED (pristine demand zone)
                if unmitBullBar == 0
                    unmitBullBar := i

        //////////////////////////////////////////
        // BEARISH OB DETECTION (Supply Zone)
        //////////////////////////////////////////

        // Step 1: Check SWEEP - OB candle swept highs of adjacent bars
        bool bearSweep = high[i] > high[i-1] and high[i] > high[i+1]

        // Step 2: Check GAP AT FORMATION (both bar[2] and bar[3] patterns)
        bool bearGap2 = low[i] > high[i-2]  // bar[2] formation pattern
        bool bearGap3 = low[i] > high[i-3]  // bar[3] formation pattern
        bool bearGapAtFormation = bearGap2 or bearGap3

        // OB formed if BOTH sweep AND gap conditions were true
        bool bearOBFormed = bearSweep and bearGapAtFormation

        if bearOBFormed
            float obHigh = high[i]
            float obLow = low[i]

            // Calculate test price (30% into OB from low)
            float testPrice = obLow + ((obHigh - obLow) * ob_testPercent / 100)

            // Step 3: Check OB lifecycle - Active → Mitigated → Breaker → Reversed
            bool isBroken = false
            bool isReversed = false
            bool hasReachedTestLevel = false
            bool isTested = false
            int startBar = bearGap3 ? i - 4 : i - 3
            int j = startBar
            while j >= 1
                if not isBroken
                    // Check if became BREAKER (close above OB's high)
                    if close[j] > obHigh
                        isBroken := true
                    else
                        // Check mitigation (two-step process from original indicator)
                        // Step 1: Price reaches test level (wicked into OB zone)
                        if not hasReachedTestLevel and high[j] >= testPrice
                            hasReachedTestLevel := true
                        // Step 2: After reaching test level, check if price returns/bounces
                        if hasReachedTestLevel and not isTested
                            if high[j] <= obHigh or (high[j] > obHigh and low[j] <= obHigh)
                                isTested := true
                else if not isReversed
                    // Check if became REVERSED (close back below OB's low)
                    if close[j] < obLow
                        isReversed := true
                j := j - 1

            // Categorize based on final state (priority: reversed > broken > mitigated > unmitigated)
            if isReversed
                // Skip - reversed OBs are no longer valid
                na(close)  // no-op: skip reversed OBs
            else if isBroken
                // It's a BREAKER SUPPORT (supply zone that failed)
                if brkSupBar == 0
                    brkSupBar := i
            else if isTested
                // It's MITIGATED (tested but not broken)
                if mitBearBar == 0
                    mitBearBar := i
            else
                // It's still UNMITIGATED (pristine supply zone)
                if unmitBearBar == 0
                    unmitBearBar := i

    [unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar]

// Combined NWE + OB scan for H4 and D1 timeframes
// Returns: [yhat, upper_far, unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar]

checkSignalWithOB() =>
    // Calculate NWE envelope (kept for future signal use)
    [yhat, upper_near, upper_far, lower_near, lower_far] = calcNWE(close, high, low, nwe_h, nwe_alpha, nwe_x0, nwe_atrLen, nwe_nearFactor, nwe_farFactor)

    // Scan 100 bars for OBs in 6 categories

    // Return NWE + OB data (using [1] offset for non-repainting NWE values)
    [yhat[1], upper_far[1], unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar]

// Helper function to format symbol name (removes exchange prefix)
getSymbolName(string fullSymbol) =>
    string result = fullSymbol
    if str.contains(fullSymbol, ":")
        result := array.get(str.split(fullSymbol, ":"), 1)
    result

// Helper function to format unmitigated OB pair as "B:x S:y"
// B = Bullish (demand), S = Bearish (supply)
// Shows bar index or "-" if none found
getUnmitPair(int bullBar, int bearBar) =>
    string bull = bullBar > 0 ? str.tostring(int(bullBar)) : "-"
    string bear = bearBar > 0 ? str.tostring(int(bearBar)) : "-"
    "B:" + bull + " S:" + bear

// Helper function to format breaker OB pair as "R:x S:y"
// R = Resistance (broken bull), S = Support (broken bear)
// Shows bar index or "-" if none found
getBrkPair(int resBar, int supBar) =>
    string res = resBar > 0 ? str.tostring(int(resBar)) : "-"
    string sup = supBar > 0 ? str.tostring(int(supBar)) : "-"
    "R:" + res + " S:" + sup

// Helper function to get color based on whether any OBs found
getUnmitColor(int bullBar, int bearBar) =>
    if bullBar > 0 and bearBar > 0
        color.lime          // Both found - green
    else if bullBar > 0 or bearBar > 0
        color.yellow        // One found - yellow
    else
        color.gray          // None found - gray

getBrkColor(int resBar, int supBar) =>
    if resBar > 0 and supBar > 0
        color.orange        // Both breakers found
    else if resBar > 0 or supBar > 0
        color.new(color.orange, 50)  // One breaker found
    else
        color.gray          // None found

// Helper function to format mitigated OB pair as "B:x S:y"
// Same format as unmit - shows bar index or "-" if none found
getMitPair(int bullBar, int bearBar) =>
    string bull = bullBar > 0 ? str.tostring(int(bullBar)) : "-"
    string bear = bearBar > 0 ? str.tostring(int(bearBar)) : "-"
    "B:" + bull + " S:" + bear

getMitColor(int bullBar, int bearBar) =>
    if bullBar > 0 and bearBar > 0
        color.aqua          // Both mitigated found
    else if bullBar > 0 or bearBar > 0
        color.new(color.aqua, 50)  // One mitigated found
    else

    for i = 4 to 30
        // Step 1: Check SWEEP
        bool bullSweep = low[i] < low[i-1] and low[i] < low[i+1]

        // Step 2: Check GAP AT FORMATION (both bar[2] and bar[3] patterns)
        bool bullGap2 = high[i] < low[i-2]
        bool bullGap3 = high[i] < low[i-3]
        bool bullGapAtFormation = bullGap2 or bullGap3

        // OB formed if BOTH conditions were true
        bool bullOBFormed = bullSweep and bullGapAtFormation

        if bullOBFormed
            float obHigh = high[i]
            float obLow = low[i]
            string gapPattern = bullGap2 ? "bar[2]" : "bar[3]"

            // Calculate test price (30% into OB from high)
            float testPrice = obHigh - ((obHigh - obLow) * ob_testPercent / 100)

            // Step 3: Check lifecycle - Active → Mitigated → Breaker → Reversed
            bool isBroken = false
            bool isReversed = false
            bool hasReachedTestLevel = false
            bool isTested = false
            int startBar = bullGap3 ? i - 4 : i - 3
            int j = startBar
            while j >= 1
                if not isBroken
                    if close[j] < obLow
                        isBroken := true
                    else
                        // Check mitigation (two-step process)
                        if not hasReachedTestLevel and low[j] <= testPrice
                            hasReachedTestLevel := true
                        if hasReachedTestLevel and not isTested
                            if low[j] >= obLow or (low[j] < obLow and high[j] >= obLow)
                                isTested := true
                else if not isReversed
                    if close[j] > obHigh
                        isReversed := true
                j := j - 1

            // Determine category
            string category = isReversed ? "REVERSED" : (isBroken ? "BREAKER_RES" : (isTested ? "MIT_BULL" : "UNMIT_BULL"))
            if not isReversed
                if isBroken and firstBrkRes == 0
                    firstBrkRes := i
                else if isTested and firstMitBull == 0
                    firstMitBull := i
                else if not isBroken and not isTested and firstUnmitBull == 0
                    firstUnmitBull := i

            // Log this OB with all details
            log.info("Bar[{0}]: FORMED ({1}) | obHigh={2,number,#.#####} obLow={3,number,#.#####} testPrice={4,number,#.#####}", i, gapPattern, obHigh, obLow, testPrice)
            log.info("  Lifecycle: tested={0} broken={1} reversed={2} => {3}", isTested, isBroken, isReversed, category)

    // Summary
    log.warning("--- SUMMARY ---")
    log.info("First Unmit Bull: bar {0}", firstUnmitBull)
    log.info("First Mit Bull: bar {0}", firstMitBull)
    log.info("First Breaker Res: bar {0}", firstBrkRes)

    // Also log what the actual scan function returned for comparison
    [scanUnmitBull, scanUnmitBear, scanMitBull, scanMitBear, scanBrkRes, scanBrkSup] = scanOBRange()
    log.warning("--- ACTUAL scanOBRange() RESULTS ---")
    log.info("unmitBullBar={0} unmitBearBar={1} mitBullBar={2} mitBearBar={3} brkResBar={4} brkSupBar={5}", scanUnmitBull, scanUnmitBear, scanMitBull, scanMitBear, scanBrkRes, scanBrkSup)
