// Kernel ATR calculation using kernel-smoothed high/low/close
kernel_atr(int length, float _high, float _low, float _close) =>
    float trueRange = na(_high[1]) ? _high - _low : math.max(math.max(_high - _low, math.abs(_high - _close[1])), math.abs(_low - _close[1]))
    ta.rma(trueRange, length)

// Calculate NWE envelope bounds
// Returns: [yhat, upper_near, upper_far, lower_near, lower_far]
calcNWE(float _close, float _high, float _low, int h, float alpha, int x0, int atrLen, float nearF, float farF) =>
    // Kernel regression on price
    float yhat_close = kernels.rationalQuadratic(_close, h, alpha, x0)
    float yhat_high = kernels.rationalQuadratic(_high, h, alpha, x0)
    float yhat_low = kernels.rationalQuadratic(_low, h, alpha, x0)

    // Kernel ATR
    float ktr = kernel_atr(atrLen, yhat_high, yhat_low, yhat_close)

    // Envelope bounds
    float upper_far = yhat_close + farF * ktr
    float upper_near = yhat_close + nearF * ktr
    float lower_near = yhat_close - nearF * ktr
    float lower_far = yhat_close - farF * ktr

    [yhat_close, upper_near, upper_far, lower_near, lower_far]

// Scan 100 bars for Order Blocks in 4 categories:
// 1. Unmitigated Bullish OB: formed with gap+sweep, not broken since (valid demand zone)
// 2. Unmitigated Bearish OB: formed with gap+sweep, not broken since (valid supply zone)
// 3. Mitigated Bullish OB: tested but not broken (demand zone that was touched)
// 4. Mitigated Bearish OB: tested but not broken (supply zone that was touched)
// 5. Breaker Resistance: Bullish OB where close broke below OB's low (broken demand → resistance)
// 6. Breaker Support: Bearish OB where close broke above OB's high (broken supply → support)
//
// Returns: [unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar, bullFVGBar, bearFVGBar]
// Each value = bars ago of nearest OB/FVG in that category (0 = none found)
//
// OB FORMATION (from OB & FVG.txt):
// An OB forms when TWO conditions are met at the same time:
// 1. SWEEP: The OB candle swept liquidity (lows for bullish, highs for bearish)
// 2. GAP: There's a gap between OB candle and validation candle (2 bars later)
//
// Bar structure at formation:
//   bar[i]   = OB candle (where sweep happened)
//   bar[i-1] = intermediate candle
//   bar[i-2] = validation candle (creates the gap)
//
// OB VALIDITY:
// - An OB stays valid until price CLOSES through its level
// - Bullish OB invalid when: any close < OB's low (becomes breaker resistance)
// - Bearish OB invalid when: any close > OB's high (becomes breaker support)
// - FVG fill is SEPARATE from OB validity (gap can close but OB still valid)

// Main scan function (non-repainting, uses confirmed bars)
// Returns: [unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar, bullFVGBar, bearFVGBar]
// OB categories:
// - unmitBullBar: Unmitigated bullish OB (pristine, never touched)
// - unmitBearBar: Unmitigated bearish OB (pristine, never touched)
// - mitBullBar: Mitigated bullish OB (tested but not broken)
// - mitBearBar: Mitigated bearish OB (tested but not broken)
// - brkResBar: Breaker resistance (broken bullish OB)
// - brkSupBar: Breaker support (broken bearish OB)
// FVG (nearest unfilled FVG from valid OBs):
// - bullFVGBar: Bar index of nearest unfilled bullish FVG (0 if none)
// - bearFVGBar: Bar index of nearest unfilled bearish FVG (0 if none)
scanOBRange() =>
    int unmitBullBar = 0
    int unmitBearBar = 0
    int mitBullBar = 0
    int mitBearBar = 0
    int brkResBar = 0
    int brkSupBar = 0
    int bullFVGBar = 0
    int bearFVGBar = 0

    // ==========================================
    // HANDLE i=2 SEPARATELY (only bar[2] pattern)
    // ==========================================
    // At i=2, we cannot check bar[3] pattern (would access low[-1])
    // So we handle it separately with only bar[2] pattern check

    // BULLISH OB at shift 2
    bool bullSweep2 = low[2] < low[1] and low[2] < low[3]
    bool bullGap2_at2 = high[2] < low[0]  // bar[2] pattern only
    if bullSweep2 and bullGap2_at2
        float obHigh2 = high[2]
        float obLow2 = low[2]
        float gapLevel2 = low[0]
        float fvgTop2 = gapLevel2
        float fvgBottom2 = obHigh2
        float fvgHeight2 = fvgTop2 - fvgBottom2
        float testPrice2 = obHigh2 - ((obHigh2 - obLow2) * ob_testPercent / 100)

        bool isBroken2 = false
        bool isReversed2 = false
        bool hasReachedTestLevel2 = false
        bool isTested2 = false
        bool isFVGFilled2 = false
        int j2 = -1  // Start after validation bar (bar[0] is validation, so start at current forming bar which doesn't exist yet)
        // No lifecycle loop needed - OB just formed, check current bar only
        // Actually for i=2, startBar = i - 3 = -1, so no bars to check yet

        // Categorize (newly formed = unmitigated)
        if unmitBullBar == 0
            unmitBullBar := 2
        if not isFVGFilled2 and bullFVGBar == 0
            bullFVGBar := 2

    // BEARISH OB at shift 2
    bool bearSweep2 = high[2] > high[1] and high[2] > high[3]
    bool bearGap2_at2 = low[2] > high[0]  // bar[2] pattern only
    if bearSweep2 and bearGap2_at2
        float obHigh2b = high[2]
        float obLow2b = low[2]
        float gapLevel2b = high[0]
        float fvgTop2b = obLow2b
        float fvgBottom2b = gapLevel2b
        float fvgHeight2b = fvgTop2b - fvgBottom2b
        float testPrice2b = obLow2b + ((obHigh2b - obLow2b) * ob_testPercent / 100)

        bool isBroken2b = false
        bool isReversed2b = false
        bool hasReachedTestLevel2b = false
        bool isTested2b = false
        bool isFVGFilled2b = false
        // No lifecycle loop needed - OB just formed

        // Categorize (newly formed = unmitigated)
        if unmitBearBar == 0
            unmitBearBar := 2
        if not isFVGFilled2b and bearFVGBar == 0
            bearFVGBar := 2

    // ==========================================
    // MAIN LOOP: i=3 to 100 (bar[3] pattern safe)
    // ==========================================
    // Now i is always >= 3, so low[i-3] is always valid (minimum low[0])
    for i = 3 to 100
        //////////////////////////////////////////
        // BULLISH OB DETECTION (Demand Zone)
        //////////////////////////////////////////

        // Step 1: Check SWEEP - OB candle swept lows of adjacent bars
        bool bullSweep = low[i] < low[i-1] and low[i] < low[i+1]

        // Step 2: Check GAP AT FORMATION
        // Original indicator checks both bar[2] and bar[3] patterns:
        // - bar[2] pattern: validation bar is 2 bars after OB → high[i] < low[i-2]
        // - bar[3] pattern: validation bar is 3 bars after OB → high[i] < low[i-3]
        bool bullGap2 = high[i] < low[i-2]  // bar[2] formation pattern
        bool bullGap3 = high[i] < low[i-3]  // bar[3] formation pattern (safe: i >= 3 in this loop)
        bool bullGapAtFormation = bullGap2 or bullGap3

        // OB formed if BOTH sweep AND gap conditions were true
        bool bullOBFormed = bullSweep and bullGapAtFormation

        if bullOBFormed
            float obHigh = high[i]
            float obLow = low[i]

            // FVG zone: gap between OB candle high and validation bar's low
            // Bullish FVG: top = gapLevel (validation bar's low), bottom = obHigh
            float gapLevel = bullGap3 ? low[i-3] : low[i-2]
            float fvgTop = gapLevel
            float fvgBottom = obHigh
            float fvgHeight = fvgTop - fvgBottom

            // Calculate test price (30% into OB from high)
            float testPrice = obHigh - ((obHigh - obLow) * ob_testPercent / 100)

            // Step 3: Check OB lifecycle - Active → Mitigated → Breaker → Reversed
            // Also track FVG fill status
            // Loop from formation to current bar, tracking state changes
            bool isBroken = false
            bool isReversed = false
            bool hasReachedTestLevel = false
            bool isTested = false
            bool isFVGFilled = false
            int startBar = bullGap3 ? i - 4 : i - 3  // Start after validation bar
            int j = startBar
            while j >= 0
                if not isBroken
                    // Check if became BREAKER (close below OB's low)
                    if close[j] < obLow
                        isBroken := true
                    else
                        // Check mitigation (two-step process from original indicator)
                        // Step 1: Price reaches test level (dipped into OB zone)
                        if not hasReachedTestLevel and low[j] <= testPrice
                            hasReachedTestLevel := true
                        // Step 2: After reaching test level, check if price returns/bounces
                        if hasReachedTestLevel and not isTested
                            if low[j] >= obLow or (low[j] < obLow and high[j] >= obLow)
                                isTested := true
                else if not isReversed
                    // Check if became REVERSED (close back above OB's high)
                    if close[j] > obHigh
                        isReversed := true

                // Check FVG fill (separate from OB lifecycle)
                // Bullish FVG filled when price wicks down into gap by fill%
                // fillLevel = gapLevel - (fvgHeight * fillPercent / 100)
                if not isFVGFilled and fvgHeight > 0
                    float fillLevel = fvgTop - (fvgHeight * fvg_fillPercent / 100)
                    if low[j] <= fillLevel
                        isFVGFilled := true

                j := j - 1

            // Categorize based on final state (priority: reversed > broken > mitigated > unmitigated)
            if isReversed
                // Skip - reversed OBs are no longer valid
                na(close)  // no-op
            else if isBroken
                // It's a BREAKER RESISTANCE (demand zone that failed)
                if brkResBar == 0
                    brkResBar := i
            else if isTested
                // It's MITIGATED (tested but not broken)
                if mitBullBar == 0
                    mitBullBar := i
            else
                // It's still UNMITIGATED (pristine demand zone)
                if unmitBullBar == 0
                    unmitBullBar := i

            // Track nearest unfilled FVG from valid OBs (not broken/reversed)
            if not isFVGFilled and not isBroken and not isReversed and bullFVGBar == 0
                bullFVGBar := i

        //////////////////////////////////////////
        // BEARISH OB DETECTION (Supply Zone)
        //////////////////////////////////////////

        // Step 1: Check SWEEP - OB candle swept highs of adjacent bars
        bool bearSweep = high[i] > high[i-1] and high[i] > high[i+1]

        // Step 2: Check GAP AT FORMATION (both bar[2] and bar[3] patterns)
        bool bearGap2 = low[i] > high[i-2]  // bar[2] formation pattern
        bool bearGap3 = low[i] > high[i-3]  // bar[3] formation pattern (safe: i >= 3 in this loop)
        bool bearGapAtFormation = bearGap2 or bearGap3

        // OB formed if BOTH sweep AND gap conditions were true
        bool bearOBFormed = bearSweep and bearGapAtFormation

        if bearOBFormed
            float obHigh = high[i]
            float obLow = low[i]

            // FVG zone: gap between validation bar's high and OB candle low
            // Bearish FVG: top = obLow, bottom = gapLevel (validation bar's high)
            float gapLevel = bearGap3 ? high[i-3] : high[i-2]
            float fvgTop = obLow
            float fvgBottom = gapLevel
            float fvgHeight = fvgTop - fvgBottom

            // Calculate test price (30% into OB from low)
            float testPrice = obLow + ((obHigh - obLow) * ob_testPercent / 100)

            // Step 3: Check OB lifecycle - Active → Mitigated → Breaker → Reversed
            // Also track FVG fill status
            bool isBroken = false
            bool isReversed = false
            bool hasReachedTestLevel = false
            bool isTested = false
            bool isFVGFilled = false
            int startBar = bearGap3 ? i - 4 : i - 3
            int j = startBar
            while j >= 0
                if not isBroken
                    // Check if became BREAKER (close above OB's high)
                    if close[j] > obHigh
                        isBroken := true
                    else
                        // Check mitigation (two-step process from original indicator)
                        // Step 1: Price reaches test level (wicked into OB zone)
                        if not hasReachedTestLevel and high[j] >= testPrice
                            hasReachedTestLevel := true
                        // Step 2: After reaching test level, check if price returns/bounces
                        if hasReachedTestLevel and not isTested
                            if high[j] <= obHigh or (high[j] > obHigh and low[j] <= obHigh)
                                isTested := true
                else if not isReversed
                    // Check if became REVERSED (close back below OB's low)
                    if close[j] < obLow
                        isReversed := true

                // Check FVG fill (separate from OB lifecycle)
                // Bearish FVG filled when price wicks up into gap by fill%
                // fillLevel = gapLevel + (fvgHeight * fillPercent / 100)
                if not isFVGFilled and fvgHeight > 0
                    float fillLevel = fvgBottom + (fvgHeight * fvg_fillPercent / 100)
                    if high[j] >= fillLevel
                        isFVGFilled := true

                j := j - 1

            // Categorize based on final state (priority: reversed > broken > mitigated > unmitigated)
            if isReversed
                // Skip - reversed OBs are no longer valid
                na(close)  // no-op
            else if isBroken
                // It's a BREAKER SUPPORT (supply zone that failed)
                if brkSupBar == 0
                    brkSupBar := i
            else if isTested
                // It's MITIGATED (tested but not broken)
                if mitBearBar == 0
                    mitBearBar := i
            else
                // It's still UNMITIGATED (pristine supply zone)
                if unmitBearBar == 0
                    unmitBearBar := i

            // Track nearest unfilled FVG from valid OBs (not broken/reversed)
            if not isFVGFilled and not isBroken and not isReversed and bearFVGBar == 0
                bearFVGBar := i

    [unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar, bullFVGBar, bearFVGBar]

checkSignalWithOB() =>

    // Return NWE + OB + FVG data (using [1] offset for non-repainting NWE values)
    [yhat[1], upper_far[1], unmitBullBar, unmitBearBar, mitBullBar, mitBearBar, brkResBar, brkSupBar, bullFVGBar, bearFVGBar]

// Helper function to format symbol name (removes exchange prefix)
getSymbolName(string fullSymbol) =>
    string result = fullSymbol
    if str.contains(fullSymbol, ":")
        result := array.get(str.split(fullSymbol, ":"), 1)
    result

// Helper function to format bar pair as "B:x S:y"
// B = Bullish, S = Bearish (Supply)
// Shows bar index or "-" if none found
formatPair(int bullBar, int bearBar) =>
    string bull = bullBar > 0 ? str.tostring(int(bullBar)) : "-"
    string bear = bearBar > 0 ? str.tostring(int(bearBar)) : "-"
    "B:" + bull + " S:" + bear

// Helper function to get color based on whether any items found
getColor(int bullBar, int bearBar, color bothColor, color oneColor) =>
    if bullBar > 0 and bearBar > 0
        bothColor
    else if bullBar > 0 or bearBar > 0
        oneColor
    else
        color.gray
